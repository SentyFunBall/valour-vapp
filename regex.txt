math:
+ (plus), -(minus/negate), *(multiply), /(divide) ^(power)
valid types:
num: (int) 123, (float) 123.456
str: "text"
bool: 0 (false) 1 (true)
lists: ["1", 2]

var assignment:
var (var name) = (value)
var i = 0
for loops:
for i = (start value) until (final value) then
    (code)
end

(end is just for multiline code)

while loops:
while (condition) then
    (code)
end

compare expressions:
>, >=. == <=, <, !=

logical operators:
and, or, not (words, not &&, ||, !)

if statements:
if (expr) then
(code)
elif (expr) then
(code)
else then
(code)

function assignment:
function (function name) ([various parameters]) -> (return)

or

function (function name) ([parameters])
    (code)
    (return)

(no classes)
